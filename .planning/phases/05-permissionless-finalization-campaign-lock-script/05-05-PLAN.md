```typescript
async function testNonCreatorFinalization() {
  console.log("=== Test: Non-Creator Permissionless Finalization ===");

  // Setup: Get two accounts (creator and non-creator)
  const creatorSigner = devnetAccounts[0];
  const nonCreatorSigner = devnetAccounts[1];

  // Step 1: Create campaign
  const campaignParams: CampaignParams = {
    // ... campaign details ...
    deadlineBlock: currentBlock + 20,  // 20 blocks from now
  };
  const createTx = await builder.createCampaign(creatorSigner, campaignParams);
  console.log(`Campaign created: ${createTx}`);

  // Step 2: Pledge from creator
  const pledgeTx = await builder.createPledgeWithReceipt(creatorSigner, pledgeParams);
  console.log(`Pledge created: ${pledgeTx}`);

  // Step 3: Non-creator attempts finalize before deadline (should fail)
  try {
    await builder.finalizeCampaign(nonCreatorSigner, {
      campaignOutPoint: { txHash: createTx, index: 0 },
      // since field will be set by transaction builder
      // current_block < deadline, so since validation should reject
    });
    console.error("ERROR: Finalization before deadline should have failed!");
    return false;
  } catch (err) {
    console.log(`✓ Expected failure (before deadline): ${err}`);
  }

  // Step 4: Wait for blocks (or skip on fast devnet)
  // In devnet, blocks may advance automatically or we can wait
  console.log("Waiting for deadline to pass...");
  await waitForBlocks(20);

  // Step 5: Non-creator finalizes after deadline (should succeed)
  const finalizeTx = await builder.finalizeCampaign(nonCreatorSigner, {
    campaignOutPoint: { txHash: createTx, index: 0 },
    campaignData: { /* loaded from indexer or previous tx */ },
    newStatus: CampaignStatus.Success,  // or Failed based on goal
  });
  console.log(`✓ Non-creator finalization succeeded: ${finalizeTx}`);

  // Step 6: Non-creator releases funds
  const releaseTx = await builder.permissionlessRelease(nonCreatorSigner, releaseParams);
  console.log(`✓ Permissionless release succeeded: ${releaseTx}`);

  // Step 7: Creator attempts finalize again (should fail)
  try {
    await builder.finalizeCampaign(creatorSigner, { /* same params */ });
    console.error("ERROR: Double finalization should have failed!");
    return false;
  } catch (err) {
    console.log(`✓ Expected failure (already finalized): ${err}`);
  }

  console.log("✓ Test passed: Non-creator permissionless finalization works!");
  return true;
}
```
